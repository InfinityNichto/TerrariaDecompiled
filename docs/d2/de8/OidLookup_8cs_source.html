20 public static string ToFriendlyName(string oid, OidGroup oidGroup, bool fallBackToAllGroups)
21 {
22 if (oid == null)
23 {
24 throw new ArgumentNullException("oid");
25 }
26 bool flag = ShouldUseCache(oidGroup);
27 if (flag && (s_oidToFriendlyName.TryGetValue(oid, out var value) || s_compatOids.TryGetValue(oid, out value) || s_lateBoundOidToFriendlyName.TryGetValue(oid, out value)))
28 {
29 return value;
30 }
31 value = NativeOidToFriendlyName(oid, oidGroup, fallBackToAllGroups);
32 if (flag && value != null)
33 {
34 s_lateBoundOidToFriendlyName.TryAdd(oid, value);
35 }
36 return value;
37 }
