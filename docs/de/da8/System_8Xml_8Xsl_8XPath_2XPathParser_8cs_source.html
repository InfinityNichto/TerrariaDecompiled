24 public Node Parse(XPathScanner scanner, IXPathBuilder<Node> builder, LexKind endLex)
25 {

28 _builder = builder;

102 private Node ParseStep()
103 {
104 Node val;
105 if (LexKind.Dot == _scanner.Kind)
106 {
107 _scanner.NextLex();
108 val = _builder.Axis(XPathAxis.Self, XPathNodeType.All, null, null);
109 if (LexKind.LBracket == _scanner.Kind)
110 {
111 throw _scanner.CreateException(System.SR.XPath_PredicateAfterDot);
112 }
113 }
114 else if (LexKind.DotDot == _scanner.Kind)
115 {
116 _scanner.NextLex();
117 val = _builder.Axis(XPathAxis.Parent, XPathNodeType.All, null, null);
118 if (LexKind.LBracket == _scanner.Kind)
119 {
120 throw _scanner.CreateException(System.SR.XPath_PredicateAfterDotDot);
121 }
122 }
123 else
124 {
125 XPathAxis axis;
126 switch (_scanner.Kind)
127 {
128 case LexKind.Axis:
129 axis = _scanner.Axis;
130 _scanner.NextLex();
131 _scanner.NextLex();
132 break;
133 case LexKind.At:
134 axis = XPathAxis.Attribute;
135 _scanner.NextLex();
136 break;
137 case LexKind.Name:
138 case LexKind.Star:
139 axis = XPathAxis.Child;
140 break;
141 default:
142 throw _scanner.CreateException(System.SR.XPath_UnexpectedToken, _scanner.RawValue);
143 }
144 val = ParseNodeTest(axis);
145 while (LexKind.LBracket == _scanner.Kind)
146 {
147 val = _builder.Predicate(val, ParsePredicate(), IsReverseAxis(axis));
148 }
149 }
150 return val;
151 }

195 internal static void InternalParseNodeTest(XPathScanner scanner, XPathAxis axis, out XPathNodeType nodeType, out string nodePrefix, out string nodeName)
196 {
197 switch (scanner.Kind)
198 {
199 case LexKind.Name:
200 if (scanner.CanBeFunction && IsNodeType(scanner))
201 {
202 nodePrefix = null;
203 nodeName = null;
204 switch (scanner.Name)
205 {
206 case "comment":
207 nodeType = XPathNodeType.Comment;
208 break;
209 case "text":
210 nodeType = XPathNodeType.Text;
211 break;
212 case "node":
213 nodeType = XPathNodeType.All;
214 break;
215 default:
216 nodeType = XPathNodeType.ProcessingInstruction;
217 break;
218 }
219 scanner.NextLex();
220 scanner.PassToken(LexKind.LParens);
221 if (nodeType == XPathNodeType.ProcessingInstruction && scanner.Kind != LexKind.RParens)
222 {
223 scanner.CheckToken(LexKind.String);
224 nodePrefix = string.Empty;
225 nodeName = scanner.StringValue;
226 scanner.NextLex();
227 }
228 scanner.PassToken(LexKind.RParens);
229 }
230 else
231 {
232 nodePrefix = scanner.Prefix;
233 nodeName = scanner.Name;
234 nodeType = PrincipalNodeType(axis);
235 scanner.NextLex();
236 if (nodeName == "*")
237 {
238 nodeName = null;
239 }
240 }
241 break;
242 case LexKind.Star:
243 nodePrefix = null;
244 nodeName = null;
245 nodeType = PrincipalNodeType(axis);
246 scanner.NextLex();
247 break;
248 default:
249 throw scanner.CreateException(System.SR.XPath_NodeTestExpected, scanner.RawValue);
250 }
251 }

266 private Node ParseSubExpr(int callerPrec)
267 {
268 if (++_parseSubExprDepth > 1024 && System.LocalAppContextSwitches.LimitXPathComplexity)
269 {
270 throw _scanner.CreateException(System.SR.Xslt_InputTooComplex);
271 }
272 Node val;
273 if (_scanner.Kind == LexKind.Minus)
274 {
275 XPathOperator xPathOperator = XPathOperator.UnaryMinus;
276 int callerPrec2 = s_XPathOperatorPrecedence[(int)xPathOperator];
277 _scanner.NextLex();
278 val = _builder.Operator(xPathOperator, ParseSubExpr(callerPrec2), default(Node));
279 }
280 else
281 {
282 val = ParseUnionExpr();
283 }
284 while (true)
285 {
286 XPathOperator xPathOperator = (XPathOperator)((_scanner.Kind <= LexKind.Union) ? _scanner.Kind : LexKind.Unknown);
287 int num = s_XPathOperatorPrecedence[(int)xPathOperator];
288 if (num <= callerPrec)
289 {
290 break;
291 }
292 _scanner.NextLex();
293 val = _builder.Operator(xPathOperator, val, ParseSubExpr(num));
294 }

374 private Node ParsePrimaryExpr()
375 {
376 Node result;
377 switch (_scanner.Kind)
378 {
379 case LexKind.String:
380 result = _builder.String(_scanner.StringValue);
381 _scanner.NextLex();
382 break;
383 case LexKind.Number:
384 result = _builder.Number(XPathConvert.StringToDouble(_scanner.RawValue));
385 _scanner.NextLex();
386 break;
387 case LexKind.Dollar:

390 _scanner.NextLex();
391 _scanner.CheckToken(LexKind.Name);
392 PushPosInfo(lexStart, _scanner.LexStart + _scanner.LexSize);
393 result = _builder.Variable(_scanner.Prefix, _scanner.Name);
394 PopPosInfo();
395 _scanner.NextLex();
396 break;
397 }
398 case LexKind.LParens:
399 _scanner.NextLex();
400 result = ParseExpr();
401 _scanner.PassToken(LexKind.RParens);
402 break;
403 default:
404 result = ParseFunctionCall();
405 break;
406 }

410 private Node ParseFunctionCall()
411 {
412 List<Node> list = new List<Node>();
413 string name = _scanner.Name;
414 string prefix = _scanner.Prefix;
415 int lexStart = _scanner.LexStart;
