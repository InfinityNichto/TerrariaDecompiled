101 {

103 {

105 }
106 AdjustDepth(state);
107 ResetRecord(state);
108 PopElementScope();

111 nspace = ((nspace != null) ? _nameTable.Add(nspace) : _atoms.Empty);

114 case XPathNodeType.Element:
