64 public override void Validate()
65 {
66 if (schemaInfo.SchemaType == SchemaType.DTD)
67 {
68 switch (reader.NodeType)
69 {
70 case XmlNodeType.Element:
71 ValidateElement();
72 if (reader.IsEmptyElement)
73 {
74 goto case XmlNodeType.EndElement;
75 }
76 break;
77 case XmlNodeType.Whitespace:
78 case XmlNodeType.SignificantWhitespace:
79 if (MeetsStandAloneConstraint())
80 {
81 ValidateWhitespace();
82 }
83 break;
84 case XmlNodeType.ProcessingInstruction:
85 case XmlNodeType.Comment:
86 ValidatePIComment();
87 break;
88 case XmlNodeType.Text:
89 case XmlNodeType.CDATA:
90 ValidateText();
91 break;
92 case XmlNodeType.EntityReference:
93 if (!GenEntity(new XmlQualifiedName(reader.LocalName, reader.Prefix)))
94 {
95 ValidateText();
96 }
97 break;
98 case XmlNodeType.EndElement:
99 ValidateEndElement();
100 break;
101 case XmlNodeType.Attribute:
102 case XmlNodeType.Entity:
103 case XmlNodeType.Document:
104 case XmlNodeType.DocumentType:
105 case XmlNodeType.DocumentFragment:
106 case XmlNodeType.Notation:
107 break;
108 }
109 }
110 else if (reader.Depth == 0 && reader.NodeType == XmlNodeType.Element)
111 {
112 SendValidationEvent(System.SR.Xml_NoDTDPresent, _name.ToString(), XmlSeverityType.Warning);
113 }
114 }

161 private void ValidateStartElement()
162 {
163 if (context.ElementDecl != null)
164 {
165 base.Reader.SchemaTypeObject = context.ElementDecl.SchemaType;
166 if (base.Reader.IsEmptyElement && context.ElementDecl.DefaultValueTyped != null)
167 {
168 base.Reader.TypedValueObject = context.ElementDecl.DefaultValueTyped;
169 context.IsNill = true;
170 }
171 if (context.ElementDecl.HasRequiredAttribute)
172 {
173 _attPresence.Clear();
174 }
175 }
176 if (!base.Reader.MoveToFirstAttribute())
177 {
178 return;
179 }
180 do
181 {
182 try
183 {
184 reader.SchemaTypeObject = null;
185 SchemaAttDef attDef = context.ElementDecl.GetAttDef(new XmlQualifiedName(reader.LocalName, reader.Prefix));
186 if (attDef != null)
187 {

195 CheckValue(base.Reader.Value, attDef);

197 }
198 else

201 }
202 }
203 catch (XmlSchemaException ex)

205 ex.SetSource(base.Reader.BaseURI, base.PositionInfo.LineNumber, base.PositionInfo.LinePosition);
206 SendValidationEvent(ex);
207 }
208 }
209 while (base.Reader.MoveToNextAttribute());

211 }

323 private void CheckValue(string value, SchemaAttDef attdef)
324 {
325 try
326 {
327 reader.TypedValueObject = null;
328 bool flag = attdef != null;
329 XmlSchemaDatatype xmlSchemaDatatype = (flag ? attdef.Datatype : context.ElementDecl.Datatype);
330 if (xmlSchemaDatatype == null)
331 {
332 return;
333 }
334 if (xmlSchemaDatatype.TokenizedType != 0)
335 {
336 value = value.Trim();
337 }
338 object obj = xmlSchemaDatatype.ParseValue(value, base.NameTable, s_namespaceManager);
339 reader.TypedValueObject = obj;
340 XmlTokenizedType tokenizedType = xmlSchemaDatatype.TokenizedType;
341 if (tokenizedType == XmlTokenizedType.ENTITY || tokenizedType == XmlTokenizedType.ID || tokenizedType == XmlTokenizedType.IDREF)
342 {
343 if (xmlSchemaDatatype.Variety == XmlSchemaDatatypeVariety.List)
344 {
345 string[] array = (string[])obj;
346 for (int i = 0; i < array.Length; i++)
347 {
348 ProcessTokenizedType(xmlSchemaDatatype.TokenizedType, array[i]);
349 }
350 }
351 else
352 {
353 ProcessTokenizedType(xmlSchemaDatatype.TokenizedType, (string)obj);
354 }
355 }
356 SchemaDeclBase schemaDeclBase = (flag ? ((SchemaDeclBase)attdef) : ((SchemaDeclBase)context.ElementDecl));
357 if (schemaDeclBase.Values != null && !schemaDeclBase.CheckEnumeration(obj))
358 {
359 if (xmlSchemaDatatype.TokenizedType == XmlTokenizedType.NOTATION)
360 {
361 SendValidationEvent(System.SR.Sch_NotationValue, obj.ToString());
362 }
363 else
364 {
365 SendValidationEvent(System.SR.Sch_EnumerationValue, obj.ToString());
366 }
367 }
368 if (!schemaDeclBase.CheckValue(obj))
369 {
370 if (flag)
371 {
372 SendValidationEvent(System.SR.Sch_FixedAttributeValue, attdef.Name.ToString());
373 }
374 else
375 {
376 SendValidationEvent(System.SR.Sch_FixedElementValue, XmlSchemaValidator.QNameString(context.LocalName, context.Namespace));
377 }
378 }
379 }
380 catch (XmlSchemaException)
381 {
382 if (attdef != null)
383 {
384 SendValidationEvent(System.SR.Sch_AttributeValueDataType, attdef.Name.ToString());
385 }
386 else
387 {
388 SendValidationEvent(System.SR.Sch_ElementValueDataType, XmlSchemaValidator.QNameString(context.LocalName, context.Namespace));
389 }
390 }

411 private bool GenEntity(XmlQualifiedName qname)
412 {
413 string name = qname.Name;
414 if (name[0] == '#')
