445 internal JsonConverter GetConverterInternal(Type typeToConvert)
446 {
447 if (_converters.TryGetValue(typeToConvert, out var value))

449 return value;
450 }

452 foreach (JsonConverter converter in Converters)
453 {
454 if (converter.CanConvert(typeToConvert))
455 {
456 value = converter;
457 break;
458 }
459 }
460 if (value == null)
461 {
462 JsonConverterAttribute jsonConverterAttribute = (JsonConverterAttribute)GetAttributeThatCanHaveMultiple(typeToConvert, typeof(JsonConverterAttribute));
463 if (jsonConverterAttribute != null)

465 value = GetConverterFromAttribute(jsonConverterAttribute, typeToConvert, typeToConvert, null);
466 }
467 }
468 if (value == null)

477 value = value2;

509 {
510 Type converterType = converterAttribute.ConverterType;

512 if (converterType == null)

518 }
519 }
