198 public X509Certificate2 Create(X509Certificate2 issuerCertificate, DateTimeOffset notBefore, DateTimeOffset notAfter, ReadOnlySpan<byte> serialNumber)
199 {
200 if (issuerCertificate == null)
201 {
202 throw new ArgumentNullException("issuerCertificate");
203 }
204 if (!issuerCertificate.HasPrivateKey)
205 {
206 throw new ArgumentException(System.SR.Cryptography_CertReq_IssuerRequiresPrivateKey, "issuerCertificate");
207 }
208 if (notAfter < notBefore)
209 {
210 throw new ArgumentException(System.SR.Cryptography_CertReq_DatesReversed);
211 }
212 if (serialNumber.IsEmpty)
213 {
214 throw new ArgumentException(System.SR.Arg_EmptyOrNullArray, "serialNumber");
215 }
216 if (issuerCertificate.PublicKey.Oid.Value != PublicKey.Oid.Value)

218 throw new ArgumentException(System.SR.Format(System.SR.Cryptography_CertReq_AlgorithmMustMatch, issuerCertificate.PublicKey.Oid.Value, PublicKey.Oid.Value), "issuerCertificate");
219 }

223 throw new ArgumentException(System.SR.Format(System.SR.Cryptography_CertReq_NotBeforeNotNested, localDateTime, issuerCertificate.NotBefore), "notBefore");
224 }
225 DateTime localDateTime2 = notAfter.LocalDateTime;
226 long ticks = localDateTime2.Ticks;
227 long num = ticks % 10000000;
228 ticks -= num;
229 localDateTime2 = new DateTime(ticks, localDateTime2.Kind);
230 if (localDateTime2 > issuerCertificate.NotAfter)
231 {
232 throw new ArgumentException(System.SR.Format(System.SR.Cryptography_CertReq_NotAfterNotNested, localDateTime2, issuerCertificate.NotAfter), "notAfter");
233 }
234 X509BasicConstraintsExtension x509BasicConstraintsExtension = (X509BasicConstraintsExtension)issuerCertificate.Extensions["2.5.29.19"];
