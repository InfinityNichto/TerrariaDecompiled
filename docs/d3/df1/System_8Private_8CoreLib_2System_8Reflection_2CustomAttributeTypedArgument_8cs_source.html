74 switch (encodedType)
75 {
76 case CustomAttributeEncoding.Undefined:
77 throw new ArgumentException(null, "encodedArg");
78 case CustomAttributeEncoding.Enum:
79 _argumentType = ResolveType(scope, encodedArg.CustomAttributeType.EnumName);
