356 private void UpdateVisualFrame()
357 {
358 float num = 0.05f;
359 float sacrificeAnimationProgress = GetSacrificeAnimationProgress();
360 float lerpValue = Utils.GetLerpValue(1f, 0.7f, sacrificeAnimationProgress, clamped: true);
361 float num2 = lerpValue * lerpValue;
362 num2 *= 2f;
363 float num3 = 1f + num2;
364 num *= num3;
365 float num4 = 1.1428572f;
366 float num5 = 1f;
367 OffsetRotationsForCogs(2f * num, _sacrificeCogsSmall);
368 OffsetRotationsForCogs(num4 * num, _sacrificeCogsMedium);
369 OffsetRotationsForCogs((0f - num5) * num, _sacrificeCogsBig);
370 int frameY = 0;
371 if (_sacrificeAnimationTimeLeft != 0)
372 {
373 float num6 = 0.1f;
374 float num7 = 1f / 15f;
375 frameY = ((sacrificeAnimationProgress >= 1f - num6) ? 8 : ((sacrificeAnimationProgress >= 1f - num6 * 2f) ? 7 : ((sacrificeAnimationProgress >= 1f - num6 * 3f) ? 6 : ((sacrificeAnimationProgress >= num7 * 4f) ? 5 : ((sacrificeAnimationProgress >= num7 * 3f) ? 4 : ((sacrificeAnimationProgress >= num7 * 2f) ? 3 : ((!(sacrificeAnimationProgress >= num7)) ? 1 : 2)))))));
376 if (_sacrificeAnimationTimeLeft == 56)
377 {
378 SoundEngine.PlaySound(63);
379 Vector2 accelerationPerFrame = new Vector2(0f, 0.16350001f);
380 for (int i = 0; i < 15; i++)
381 {
382 Vector2 initialVelocity = Main.rand.NextVector2Circular(4f, 3f);
383 if (initialVelocity.Y > 0f)
384 {
385 initialVelocity.Y = 0f - initialVelocity.Y;
386 }
387 initialVelocity.Y -= 2f;
388 _pistonParticleSystem.AddParticle(new CreativeSacrificeParticle(_pistonParticleAsset, null, initialVelocity, Vector2.Zero)
389 {
390 AccelerationPerFrame = accelerationPerFrame,
391 ScaleOffsetPerFrame = -1f / 60f
392 });
393 }
394 }
395 if (_sacrificeAnimationTimeLeft == 40 && _researchComplete)
396 {
397 _researchComplete = false;
398 SoundEngine.PlaySound(64);
399 }
400 }
401 _sacrificePistons.SetFrame(1, 9, 0, frameY, 0, 0);

473 public void SacrificeWhatYouCan()
474 {
475 Main.CreativeMenu.GetSacrificeNumbers(out var itemIdChecked, out var amountWeHave, out var amountNeededTotal);
476 int amountWeSacrificed;
477 switch (Main.CreativeMenu.SacrificeItem(out amountWeSacrificed))
478 {
479 case CreativeUI.ItemSacrificeResult.SacrificedAndDone:
480 _researchComplete = true;
481 BeginSacrificeAnimation();
482 RememberItemSacrifice(itemIdChecked, amountWeHave + amountWeSacrificed, amountNeededTotal);
483 break;
484 case CreativeUI.ItemSacrificeResult.SacrificedButNotDone:
485 _researchComplete = false;
486 BeginSacrificeAnimation();
487 RememberItemSacrifice(itemIdChecked, amountWeHave + amountWeSacrificed, amountNeededTotal);
488 break;
489 }
