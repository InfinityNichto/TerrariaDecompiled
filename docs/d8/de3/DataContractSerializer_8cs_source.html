325 internal void InternalWriteObjectContent(XmlWriterDelegator writer, object graph, DataContractResolver dataContractResolver)
326 {
327 if (MaxItemsInObjectGraph == 0)
328 {
329 throw DiagnosticUtility.ExceptionUtility.ThrowHelperError(XmlObjectSerializer.CreateSerializationException(System.SR.Format(System.SR.ExceededMaxItemsQuota, MaxItemsInObjectGraph)));
330 }
331 DataContract rootContract = RootContract;
332 Type underlyingType = rootContract.UnderlyingType;
333 Type objType = ((graph == null) ? underlyingType : graph.GetType());
334 if (_serializationSurrogateProvider != null)
335 {
336 graph = SurrogateToDataContractType(_serializationSurrogateProvider, graph, underlyingType, ref objType);
337 }
338 if (dataContractResolver == null)
339 {
340 dataContractResolver = DataContractResolver;
341 }
342 if (graph == null)

344 if (IsRootXmlAny(_rootName, rootContract))
