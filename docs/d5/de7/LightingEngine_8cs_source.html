102 TimeLogger.LightingTime(2, _timer.Elapsed.TotalMilliseconds);
103 break;
104 case EngineState.Scan:
105 ProcessScan(area);
106 TimeLogger.LightingTime(3, _timer.Elapsed.TotalMilliseconds);
107 break;
108 case EngineState.Blur:

112 break;
113 }
114 IncrementState();
