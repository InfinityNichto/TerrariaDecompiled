146 public void ValidateText()
147 {
148 if (!context.NeedValidateChildren)
149 {
150 return;
151 }
152 if (context.IsNill)
153 {
154 SendValidationEvent(System.SR.Sch_ContentInNill, XmlSchemaValidator.QNameString(context.LocalName, context.Namespace));
155 return;
156 }
157 ContentValidator contentValidator = context.ElementDecl.ContentValidator;
158 switch (contentValidator.ContentType)
159 {
160 case XmlSchemaContentType.ElementOnly:
161 {
162 ArrayList arrayList = contentValidator.ExpectedElements(context, isRequiredOnly: false);
163 if (arrayList == null)
164 {
165 SendValidationEvent(System.SR.Sch_InvalidTextInElement, XmlSchemaValidator.BuildElementName(context.LocalName, context.Namespace));
166 break;
167 }
168 SendValidationEvent(System.SR.Sch_InvalidTextInElementExpecting, new string[2]
169 {
170 XmlSchemaValidator.BuildElementName(context.LocalName, context.Namespace),
171 XmlSchemaValidator.PrintExpectedElements(arrayList, getParticles: false)
172 });
173 break;
174 }
175 case XmlSchemaContentType.Empty:
176 SendValidationEvent(System.SR.Sch_InvalidTextInEmpty, string.Empty);
177 break;
178 }
179 if (checkDatatype)
180 {
181 SaveTextValue(reader.Value);
182 }
183 }

185 public void ValidateWhitespace()
186 {
187 if (context.NeedValidateChildren)
188 {
189 XmlSchemaContentType contentType = context.ElementDecl.ContentValidator.ContentType;
190 if (context.IsNill)
191 {
192 SendValidationEvent(System.SR.Sch_ContentInNill, XmlSchemaValidator.QNameString(context.LocalName, context.Namespace));
193 }
194 if (contentType == XmlSchemaContentType.Empty)
195 {
196 SendValidationEvent(System.SR.Sch_InvalidWhitespaceInEmpty, string.Empty);
