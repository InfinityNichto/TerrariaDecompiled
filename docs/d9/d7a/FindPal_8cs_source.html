207 internal static void ValidateOidValue(string keyValue)
208 {
209 if (keyValue == null)
210 {
211 throw new ArgumentNullException("keyValue");
212 }
213 int length = keyValue.Length;
214 if (length < 2)
215 {
216 throw new ArgumentException(System.SR.Argument_InvalidOidValue);
217 }
218 char c = keyValue[0];
219 if (c != '0' && c != '1' && c != '2')
220 {
221 throw new ArgumentException(System.SR.Argument_InvalidOidValue);
222 }
223 if (keyValue[1] != '.' || keyValue[length - 1] == '.')
224 {
225 throw new ArgumentException(System.SR.Argument_InvalidOidValue);
226 }
227 for (int i = 1; i < length; i++)
228 {
229 if (!char.IsDigit(keyValue[i]) && (keyValue[i] != '.' || keyValue[i + 1] == '.'))
230 {
231 throw new ArgumentException(System.SR.Argument_InvalidOidValue);
232 }
233 }
