314 DataflowMessageStatus ITargetBlock<T>.OfferMessage(DataflowMessageHeader messageHeader, T messageValue, ISourceBlock<T> source, bool consumeToAccept)
315 {
316 if (!messageHeader.IsValid)
317 {
318 throw new ArgumentException(System.SR.Argument_InvalidMessageHeader, "messageHeader");
319 }
320 if (source == null && consumeToAccept)
321 {
322 throw new ArgumentException(System.SR.Argument_CantConsumeFromANullSource, "consumeToAccept");
323 }
324 lock (_sharedResources.IncomingLock)
325 {
326 if (_decliningPermanently || _sharedResources._decliningPermanently)
327 {
328 _sharedResources.CompleteBlockIfPossible();
329 return DataflowMessageStatus.DecliningPermanently;
330 }
331 if (_sharedResources._dataflowBlockOptions.Greedy && (_sharedResources._boundingState == null || ((_sharedResources._boundingState.CountIsLessThanBound || !HasTheHighestNumberOfMessagesAvailable) && _nonGreedy.PostponedMessages.Count == 0 && _sharedResources._taskForInputProcessing == null)))
332 {
333 if (consumeToAccept)
334 {
335 messageValue = source.ConsumeMessage(messageHeader, this, out var messageConsumed);
336 if (!messageConsumed)
337 {
338 return DataflowMessageStatus.NotAvailable;
339 }
340 }
341 if (_sharedResources._boundingState != null && HasTheHighestNumberOfMessagesAvailable)
342 {
343 _sharedResources._boundingState.CurrentCount++;
344 }
345 _messages.Enqueue(messageValue);
346 CompleteIfLastJoinIsFeasible();
347 if (_sharedResources.AllTargetsHaveAtLeastOneMessage)
348 {
349 _sharedResources._joinFilledAction();
350 _sharedResources._joinsCreated++;
351 }
352 _sharedResources.CompleteBlockIfPossible();
353 return DataflowMessageStatus.Accepted;

355 if (source != null)
