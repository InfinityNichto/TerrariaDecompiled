251 protected DataContract ResolveDataContractFromKnownTypes(string typeName, string typeNs, DataContract memberTypeContract, Type declaredType)
252 {
253 XmlQualifiedName xmlQualifiedName = new XmlQualifiedName(typeName, typeNs);
254 DataContract dataContract;
255 if (_dataContractResolver == null)
256 {
257 dataContract = ResolveDataContractFromKnownTypes(xmlQualifiedName);
258 }
259 else
260 {
261 Type type = _dataContractResolver.ResolveName(typeName, typeNs, declaredType, KnownTypeResolver);
262 dataContract = ((type == null) ? null : GetDataContract(type));
263 }
264 if (dataContract == null)
265 {
266 if (memberTypeContract != null && !memberTypeContract.UnderlyingType.IsInterface && memberTypeContract.StableName == xmlQualifiedName)
267 {
268 dataContract = memberTypeContract;
269 }
270 if (dataContract == null && rootTypeDataContract != null)
271 {
272 dataContract = ((!(rootTypeDataContract.StableName == xmlQualifiedName)) ? ResolveDataContractFromRootDataContract(xmlQualifiedName) : rootTypeDataContract);
273 }
274 }
275 return dataContract;
